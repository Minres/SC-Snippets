194 char const* is_valid_msg<axi::ace_extension>(axi::ace_extension* ext){
195  auto offset = to_int(ext->get_snoop());
196  if(offset<32){ // a read access
197  if(!rd_valid[offset&0xf][to_int(ext->get_domain())])
198  return "illegal read snoop value";
199  } else {
200  if(!wr_valid[offset&0xf][to_int(ext->get_domain())])
201  return "illegal write snoop value";
202  }
203  if((ext->get_snoop()==snoop_e::READ_NO_SNOOP || ext->get_snoop()==snoop_e::WRITE_NO_SNOOP) &&
204  ext->get_domain()!=domain_e::NON_SHAREABLE && ext->get_domain()!=domain_e::SYSTEM){
205  return "illegal domain for no snoop access";
206  }
207  if((ext->get_barrier()==bar_e::MEMORY_BARRIER || ext->get_barrier()==bar_e::SYNCHRONISATION_BARRIER) && (offset&0xf)!=0)
208  return "illegal barrier/snoop value combination";
209  switch(ext->get_cache()){
210  case 4:
211  case 5:
212  case 8:
213  case 9:
214  case 12:
215  case 13:
216  return "illegal cache value";
217  }
218  return nullptr;
219 }
