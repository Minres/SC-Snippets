70 template <typename T, unsigned CHUNK_SIZE> inline void* pool_allocator<T, CHUNK_SIZE>::allocate() {
71  std::lock_guard<std::mutex> lk(payload_mtx);
72  if(!free_list.size())
73  resize();
74  auto ret = free_list.back();
75  free_list.pop_back();
76  memset(ret, 0, sizeof(T));
77  return ret;
78 }

86 template <typename T, unsigned CHUNK_SIZE> inline void pool_allocator<T, CHUNK_SIZE>::resize() {
87  auto* chunk = new std::array<chunk_type, CHUNK_SIZE>();
88  chunks.push_back(chunk);
89  for(auto& p : *chunk)
90  free_list.push_back(&p[0]);
