127  bool write(const uint8_t* data, size_t length, uint64_t offset = 0,
128  sc_core::sc_time d = sc_core::SC_ZERO_TIME) override {
129  assert("Access out of range" && offset + length <= sizeof(DATATYPE));
130  auto temp(storage);
131  auto beg = reinterpret_cast<uint8_t*>(&temp) + offset;
132  std::copy(data, data + length, beg);
133  if(wr_cb)
134  return wr_cb(*this, temp, d);
135  storage = (temp & wrmask) | (storage & ~wrmask);
136  return true;
137  }
148  bool read(uint8_t* data, size_t length, uint64_t offset = 0,
149  sc_core::sc_time d = sc_core::SC_ZERO_TIME) const override {
150  assert("Access out of range" && offset + length <= sizeof(DATATYPE));
151  auto temp(storage);
152  if(rd_cb) {
153  if(!rd_cb(*this, temp, d))
154  return false;
155  } else
156  temp &= rdmask;
157  auto beg = reinterpret_cast<uint8_t*>(&temp) + offset;
158  std::copy(beg, beg + length, data);
159  return true;
160  }
172  bool write_dbg(const uint8_t* data, size_t length, uint64_t offset = 0) override {
173  assert("Offset out of range" && offset == 0);
174  if(length != sizeof(DATATYPE))
175  return false;
176  storage = *reinterpret_cast<const DATATYPE*>(data);
177  return true;
178  }
