108 template <unsigned long long SIZE, unsigned BUSWIDTH>
109 int memory<SIZE, BUSWIDTH>::handle_operation(tlm::tlm_generic_payload& trans) {
110  sc_dt::uint64 adr = trans.get_address();
111  uint8_t* ptr = trans.get_data_ptr();
112  unsigned len = trans.get_data_length();
113  uint8_t* byt = trans.get_byte_enable_ptr();
114  unsigned wid = trans.get_streaming_width();
115  // check address range and check for unsupported features,
116  // i.e. byte enables, streaming, and bursts
117  // Can ignore DMI hint and extensions
118  if(adr + len > sc_dt::uint64(SIZE)) {
119  SC_REPORT_ERROR("TLM-2", "generic payload transaction exceeeds memory size");
120  trans.set_response_status(tlm::TLM_ADDRESS_ERROR_RESPONSE);
121  return 0;
122  }
123  if(adr + len > sc_dt::uint64(SIZE) || byt != 0 || wid < len) {
124  SC_REPORT_ERROR("TLM-2", "generic payload transaction not supported");
125  trans.set_response_status(tlm::TLM_GENERIC_ERROR_RESPONSE);
126  return 0;
127  }
128 
129  tlm::tlm_command cmd = trans.get_command();
130  SCCTRACE(SCMOD) << (cmd == tlm::TLM_READ_COMMAND ? "read" : "write") << " access to addr 0x" << std::hex << adr;
131  if(cmd == tlm::TLM_READ_COMMAND) {
132  if(mem.is_allocated(adr)) {
133  const auto& p = mem(adr / mem.page_size);
134  auto offs = adr & mem.page_addr_mask;
135  std::copy(p.data() + offs, p.data() + offs + len, ptr);
136  } else {
137  // no allocated page so return randomized data
138  for(size_t i = 0; i < len; i++)
139  ptr[i] = scc::MT19937::uniform() % 256;
140  }
141  } else if(cmd == tlm::TLM_WRITE_COMMAND) {
142  auto& p = mem(adr / mem.page_size);
143  auto offs = adr & mem.page_addr_mask;
144  std::copy(ptr, ptr + len, p.data() + offs);
145  }
146  trans.set_response_status(tlm::TLM_OK_RESPONSE);
147  trans.set_dmi_allowed(true);
148  return len;
149 }

152 inline bool memory<SIZE, BUSWIDTH>::handle_dmi(tlm::tlm_generic_payload& gp, tlm::tlm_dmi& dmi_data) {
153  auto& p = mem(gp.get_address() / mem.page_size);
154  dmi_data.set_start_address(gp.get_address() & ~mem.page_addr_mask);
155  // TODO: fix to provide the correct end address
156  dmi_data.set_end_address(dmi_data.get_start_address() + mem.page_size - 1);
157  dmi_data.set_dmi_ptr(p.data());
158  dmi_data.set_granted_access(tlm::tlm_dmi::DMI_ACCESS_READ_WRITE);
159  return true;
160 }
