98 void scc::configurer::dump_configuration(sc_core::sc_object* obj, Json::Value& parent) {
99  auto mod = dynamic_cast<sc_core::sc_module*>(obj);
100  Json::Value node{Json::objectValue};
101  for(sc_core::sc_attr_base* attr_base : obj->attr_cltn()) {
102  check_n_assign<int>(node, attr_base) || check_n_assign<unsigned>(node, attr_base) ||
103  check_n_assign<long, Json::Int64>(node, attr_base) ||
104  check_n_assign<unsigned long, Json::UInt64>(node, attr_base) ||
105  check_n_assign<long long, Json::Int64>(node, attr_base) ||
106  check_n_assign<unsigned long long, Json::UInt64>(node, attr_base) ||
107  check_n_assign<bool>(node, attr_base) ||
108  check_n_assign<float>(node, attr_base) || check_n_assign<double>(node, attr_base) ||
109  check_n_assign<std::string>(node, attr_base) || check_n_assign<char*>(node, attr_base);
110  }
111 #ifdef HAS_CCI
112  const std::string hier_name{obj->name()};
113  cci::cci_param_predicate pred{[hier_name](cci::cci_param_untyped_handle h) -> bool {
114  std::string h_name{h.name()};
115  auto sep = hier_name.length();
116  if(h_name.length() > hier_name.length()) {
117  auto path_match = h_name.compare(0, sep, hier_name) == 0; // begins with hier_name
118  auto sep_match = h_name[sep] == '.'; // has additional part
119  auto tail_nomatch = h_name.substr(sep + 1).find('.') == h_name.npos; // but no other hierarchy separator
120  return path_match && sep_match && tail_nomatch;
121  } else
122  return false;
123  }};
124  auto log_lvl_name = hier_name + ".log_level";
125  auto handles = cci_broker.get_param_handles(pred);
126  auto log_lvl_set = false;
127  for(auto& h : handles) {
128  auto value = h.get_cci_value();
129  auto paramname = std::string{h.name()};
130  if(paramname == log_lvl_name)
131  log_lvl_set = true;
132  auto basename = paramname.substr(paramname.find_last_of('.') + 1);
133  if(value.is_bool())
134  node[basename] = value.get_bool();
135  else if(value.is_int())
136  node[basename] = value.get_int();
137  else if(value.is_int64())
138  node[basename] = static_cast<int64_t>(value.get_int64());
139  else if(value.is_uint())
140  node[basename] = value.get_uint();
141  else if(value.is_uint64())
142  node[basename] = static_cast<uint64_t>(value.get_uint64());
143  else if(value.is_double())
144  node[basename] = value.get_double();
145  else if(value.is_string())
146  node[basename] = value.get_string().c_str();
147  }
148  if(!log_lvl_set && mod) {
149  auto val = cci_broker.get_preset_cci_value(log_lvl_name);
150  auto global_verb = static_cast<int>(scc::get_logging_level());
151  node["log_level"] = val.is_int() ? val.get_int() : global_verb;
152  }
153 #endif
154  for(auto* o : get_sc_objects(obj))
155  dump_configuration(o, node);
156  if(!node.empty())
157  parent[obj->basename()] = node;
158 }

228 void scc::configurer::configure_cci_hierarchical(const Json::Value& node, std::string prefix) {
229  if(node.size() > 0) {
230  for(Json::Value::const_iterator itr = node.begin(); itr != node.end(); ++itr) {
231  if(!itr.key().isString())
232  return;
233  std::string key_name = itr.key().asString();
234  std::string hier_name{prefix.size() ? prefix + "." + key_name : key_name};
235  Json::Value val = node[key_name];
236  if(val.isNull() || val.isArray())
237  return;
238  else if(val.isObject())
239  configure_cci_hierarchical(*itr, hier_name);
240 #ifdef HAS_CCI
241  else {
242  cci::cci_param_handle param_handle = cci_broker.get_param_handle(hier_name);
243  if(param_handle.is_valid()) {
244  if(val.isString()) {
245  param_handle.set_cci_value(cci::cci_value(val.asString()));
246  } else if(val.isBool()) {
247  param_handle.set_cci_value(cci::cci_value(val.asBool()));
248  } else if(val.isInt()) {
249  param_handle.set_cci_value(cci::cci_value(val.asInt()));
250  } else if(val.isInt64()) {
251  param_handle.set_cci_value(cci::cci_value(val.asInt64()));
252  } else if(val.isUInt()) {
253  param_handle.set_cci_value(cci::cci_value(val.asUInt()));
254  } else if(val.isUInt64()) {
255  param_handle.set_cci_value(cci::cci_value(val.asUInt64()));
256  } else if(val.isDouble()) {
257  param_handle.set_cci_value(cci::cci_value(val.asDouble()));
258  }
259  } else {
260  if(val.isString()) {
261  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asString()));
262  } else if(val.isBool()) {
263  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asBool()));
264  } else if(val.isInt()) {
265  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asInt()));
266  } else if(val.isInt64()) {
267  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asInt64()));
268  } else if(val.isUInt()) {
269  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asUInt()));
270  } else if(val.isUInt64()) {
271  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asUInt64()));
272  } else if(val.isDouble()) {
273  cci_broker.set_preset_cci_value(hier_name, cci::cci_value(val.asDouble()));

282 void scc::configurer::check_config_hierarchical(Json::Value const& node, std::string const& prefix) {
283  if(node.size() > 0) {
284  for(Json::Value::const_iterator itr = node.begin(); itr != node.end(); ++itr) {
285  if(!itr.key().isString())
286  return;
287  std::string key_name = itr.key().asString();
288  if(key_name == "log_level")
289  continue; // virtual attribute
290  std::string hier_name{prefix.size() ? prefix + "." + key_name : key_name};
291  Json::Value val = node[key_name];
292  if(val.isNull() || val.isArray())
293  continue;
294  else if(val.isObject()) {
295  if(!sc_core::sc_find_object(hier_name.c_str()))
296  SCCFATAL(this->name()) << "Illegal hierarchy name: '" << hier_name << "'";
297  check_config_hierarchical(*itr, hier_name);
298  }
299 #ifdef HAS_CCI
300  else {
301  auto* obj = sc_core::sc_find_object(prefix.c_str());
302  auto* attr = obj->get_attribute(key_name);
303  if(!attr) {
304  cci::cci_param_handle param_handle = cci_broker.get_param_handle(hier_name);
305  if(!param_handle.is_valid()) {
306  SCCFATAL(this->name()) << "Illegal parameter name: '" << hier_name << "'";
307  }
308  }
309  }
310 #endif
311  }
312  }
