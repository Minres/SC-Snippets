187 router<BUSWIDTH>::router(const sc_core::sc_module_name& nm, unsigned slave_cnt, unsigned master_cnt)
188 : sc_module(nm)
189 , target("target", master_cnt)
190 , initiator("intor", slave_cnt)
191 , ibases(master_cnt)
192 , tranges(slave_cnt)
193 , mutexes(slave_cnt)
194 , addr_decoder(std::numeric_limits<unsigned>::max()) {
195  for(size_t i = 0; i < target.size(); ++i) {
196  target[i].register_b_transport([=](tlm::tlm_generic_payload& trans, sc_core::sc_time& delay) -> void {

198  });
199  target[i].register_get_direct_mem_ptr([=](tlm::tlm_generic_payload& trans, tlm::tlm_dmi& dmi_data) -> bool {
200  return this->get_direct_mem_ptr(i, trans, dmi_data);
201  });
202  target[i].register_transport_dbg(
203  [=](tlm::tlm_generic_payload& trans) -> unsigned { return this->transport_dbg(i, trans); });
204  ibases[i] = 0ULL;
205  }
206  for(size_t i = 0; i < initiator.size(); ++i) {
207  initiator[i].register_invalidate_direct_mem_ptr(
208  [=](sc_dt::uint64 start_range, sc_dt::uint64 end_range) -> void {
209  this->invalidate_direct_mem_ptr(i, start_range, end_range);

211  tranges[i].base = 0ULL;

213  tranges[i].remap = false;

246 template <unsigned BUSWIDTH>
247 void router<BUSWIDTH>::b_transport(int i, tlm::tlm_generic_payload& trans, sc_core::sc_time& delay) {
248  sc_dt::uint64 address = trans.get_address();
249  if(ibases[i]) {
250  address += ibases[i];
251  trans.set_address(address);
252  }
253  size_t idx = addr_decoder.getEntry(address);
254  if(idx == addr_decoder.null_entry) {
255  if(default_idx == std::numeric_limits<size_t>::max()) {
256  trans.set_response_status(tlm::TLM_ADDRESS_ERROR_RESPONSE);
257  return;
258  }
259  idx = default_idx;
260  } else {
261  // Modify address within transaction
262  trans.set_address(address - (tranges[idx].remap ? tranges[idx].base : 0));
263  }
264  // Forward transaction to appropriate target
265  mutexes[idx].lock();
266  initiator[idx]->b_transport(trans, delay);
267  mutexes[idx].unlock();
268 }

271  sc_dt::uint64 address = trans.get_address();
272  if(ibases[i]) {
273  address += ibases[i];
274  trans.set_address(address);

279  trans.set_response_status(tlm::TLM_ADDRESS_ERROR_RESPONSE);
280  return false;

282  idx = default_idx;
283  } else {
284  // Modify address within transaction
285  trans.set_address(address - (tranges[idx].remap ? tranges[idx].base : 0));
286  }
287  bool status = initiator[idx]->get_direct_mem_ptr(trans, dmi_data);
288  // Calculate DMI address of target in system address space
289  auto offset = tranges[idx].remap ? tranges[idx].base : 0;
290  dmi_data.set_start_address(dmi_data.get_start_address() - ibases[i] + offset);
291  dmi_data.set_end_address(dmi_data.get_end_address() - ibases[i] + offset);
292  return status;
293 }

315 void router<BUSWIDTH>::invalidate_direct_mem_ptr(int id, sc_dt::uint64 start_range, sc_dt::uint64 end_range) {
316  // Reconstruct address range in system memory map
317  sc_dt::uint64 bw_start_range = start_range;
318  if(tranges[id].remap)
319  bw_start_range += tranges[id].base;
320  sc_dt::uint64 bw_end_range = end_range;
321  if(tranges[id].remap)
322  bw_end_range += tranges[id].base;
323  for(size_t i = 0; i < target.size(); ++i) {
324  target[i]->invalidate_direct_mem_ptr(bw_start_range - ibases[i], bw_end_range - ibases[i]);
325  }
