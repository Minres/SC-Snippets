49 template <unsigned int bit, unsigned int width, typename T>
50 inline constexpr typename std::make_signed<T>::type signed_bit_sub(T v) {
51 #if __cplusplus < 201402L
52  return ((v << (sizeof(T) * 8 - bit - width)) >> (sizeof(T) * 8 - width));
53 #else
54  typename std::make_signed<T>::type r = v << (sizeof(T) * 8 - bit - width);
55  typename std::make_signed<T>::type ret = (r >> (sizeof(T) * 8 - width));

128 template <typename Range, typename Value = typename Range::value_type>
129 std::string join(Range const& elements, char const* const delimiter) {
130  std::ostringstream os;
131  auto b = std::begin(elements);
132  auto e = std::end(elements);
133  if(b != e) {
134  std::copy(b, std::prev(e), std::ostream_iterator<Value>(os, delimiter));
135  b = std::prev(e);
136  }
137  if(b != e)
138  os << *b;
139  return os.str();
140 }

195 inline bool iequals(const std::string& a, const std::string& b) {
196 #if __cplusplus < 201402L
197  auto sz = a.size();
198  if(b.size() != sz)
199  return false;
200  for(auto i = 0U; i < sz; ++i)

202  return false;
203  return true;
204 #else
205  return std::equal(a.begin(), a.end(), b.begin(), b.end(), [](char a, char b) { return tolower(a) == tolower(b); });

217 inline std::string padded(std::string str, size_t width, bool show_ellipsis = true) {
