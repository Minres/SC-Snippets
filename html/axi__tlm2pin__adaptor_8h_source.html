135 template <unsigned int BUSWIDTH, unsigned int ADDRWIDTH, unsigned int IDWIDTH, unsigned int USERWIDTH>
136 inline tlm::tlm_sync_enum
137 axi_tlm2pin_adaptor<BUSWIDTH, ADDRWIDTH, IDWIDTH, USERWIDTH>::nb_transport_fw(payload_type& trans, phase_type& phase,
138  sc_core::sc_time& t) {
139  if(trans.has_mm())
140  trans.acquire();
141 
142  auto axi_id = axi::get_axi_id(trans);
143  SCCTRACE(SCMOD) << phase << " FW " << (trans.is_read() ? "RD" : "WR") << " forward trans " << std::hex << &trans
144  << std::dec << " (axi_id:" << axi_id << ")";
145  tlm::tlm_sync_enum status{tlm::TLM_ACCEPTED};
146 
147  if(phase == axi::BEGIN_PARTIAL_REQ || phase == tlm::BEGIN_REQ) {
148  auto ext = trans.get_extension<axi::axi4_extension>();
149  sc_assert(ext && "axi4_extension missing");
150 
151  if(trans.is_read()) {
152  register_trans(axi_id, trans, phase);
153  auto th = std::make_shared<trans_handle>();
154  th->payload = &trans;
155  th->phase = tlm::END_REQ;
156  active_ar_end_transactions.push_back(th);
157  sc_core::sc_spawn([this]()->void {end_ar_req();});
158  } else { // Write transaction
159  auto it = active_w_transactions.find(axi_id);
160  if(it == active_w_transactions.end())
161  register_trans(axi_id, trans, phase);
162  else {
163  auto act_trans = it->second.front();
164  if(act_trans->payload == &trans) // if transaction is ongoing update the phase
165  act_trans->phase = phase;
166  else // otherwise push transaction in the queue
167  register_trans(axi_id, trans, phase);
168  }
169  auto th = std::make_shared<trans_handle>();
170  th->payload = &trans;
171  if(phase == axi::BEGIN_PARTIAL_REQ) th->phase = axi::END_PARTIAL_REQ;
172  else th->phase = tlm::END_REQ;
173  active_aw_end_transactions.push_back(th);
174  sc_core::sc_spawn([this]()->void {end_aw_req();});
175  }
176  return status;
177  };
178  if(phase == axi::END_PARTIAL_RESP) {
179  return status;
180  };
181  if(phase == tlm::END_RESP) {
182  trans.set_response_status(tlm::TLM_OK_RESPONSE);
183  remove_trans(axi_id, trans.is_read());
184  if(trans.has_mm())
185  trans.release();
186  status = tlm::TLM_ACCEPTED;
187  return status;
188  }
189  SCCWARN(SCMOD) << phase << " unkown phase transaction combination for trans";
190  return status;
191 }

216 template <unsigned int BUSWIDTH, unsigned int ADDRWIDTH, unsigned int IDWIDTH, unsigned int USERWIDTH>
217 inline void axi_tlm2pin_adaptor<BUSWIDTH, ADDRWIDTH, IDWIDTH, USERWIDTH>::bus_thread() {
218  auto delay = sc_core::SC_ZERO_TIME;
219 
220  if(!resetn_i.read()) { // active-low reset
221  ar_valid_o.write(false);
222  aw_valid_o.write(false);
223  w_valid_o.write(false);
224  w_last_o.write(false);
225  b_ready_o.write(false);
226  r_ready_o.write(false);
227  } else {
228  if(aw_ready_i.read())
229  aw_valid_o.write(false);
230  if(w_ready_i.read()) {
231  w_valid_o.write(false);
232  w_last_o.write(false);
233  }
234  if(ar_ready_i.read()) {
235  ar_valid_o.write(false);
236  }
237  r_ready_o.write(true);
238  b_ready_o.write(true);
239 
240  // AR channel
241  for(auto& it : active_r_transactions) {
242  auto read_trans = it.second.front();
243  if(read_trans->phase == tlm::BEGIN_REQ) {
244  payload_type* p = read_trans->payload;
245  auto ext = p->get_extension<axi::axi4_extension>();
246  sc_assert(ext && "axi4_extension missing");
247 
248  ar_addr_o.write(p->get_address());
249  ar_id_o.write(ext->get_id());
250  ar_len_o.write(ext->get_length());
251  ar_size_o.write(ext->get_size());
252  ar_burst_o.write(axi::to_int(ext->get_burst()));
253  ar_lock_o.write(ext->is_exclusive());
254  ar_cache_o.write(ext->get_cache());
255  ar_prot_o.write(ext->get_prot());
256  ar_qos_o.write(ext->get_qos());
257  ar_region_o.write(ext->get_region());
258  if(ar_user_o.get_interface()) // optional user interface
259  ar_user_o->write(ext->get_user(common::id_type::CTRL));
260  ar_valid_o.write(true);
261 
262  read_trans->phase = tlm::END_REQ;
263  SCCTRACE(SCMOD) << read_trans->phase << " AR bit assignment trans: " << p << " addr: " << p->get_address()
264  << " axi_id: " << ext->get_id();
265  break;
266  }
267  }
268 
269  // AW+W channel
270  for(auto& it : active_w_transactions) {
271  auto write_trans = it.second.front();
272  if(write_trans->phase == axi::BEGIN_PARTIAL_REQ || write_trans->phase == tlm::BEGIN_REQ) {
273  payload_type* p = write_trans->payload;
274  auto ext = p->get_extension<axi::axi4_extension>();
275  sc_assert(ext && "axi4_extension missing");
276 
277  if(write_trans->beat_cnt == 0) {
278  aw_id_o.write(ext->get_id());
279  aw_addr_o.write(p->get_address());
280  aw_len_o.write(ext->get_length());
281  aw_size_o.write(ext->get_size());
282  aw_burst_o.write(axi::to_int(ext->get_burst()));
283  aw_lock_o.write(ext->is_exclusive());
284  aw_cache_o.write(ext->get_cache());
285  aw_prot_o.write(ext->get_prot());
286  aw_qos_o.write(ext->get_qos());
287  aw_region_o.write(ext->get_region());
288  if(aw_user_o.get_interface())
289  aw_user_o->write(ext->get_user(common::id_type::CTRL));
290  aw_valid_o.write(true);
291  }
292  write_trans->beat_cnt++;
293 
294  // write data
295  sc_dt::sc_biguint<BUSWIDTH> data{0};
296  for(size_t j = 0, k = 0, offset = 0; k < 32 / 8; j += 8, ++k, ++offset)
297  data.range(j + 7, j) = *(uint8_t*)(p->get_data_ptr() + offset);
298 
299  w_strb_o.write(p->get_byte_enable_length());
300  w_data_o.write(data);
301  w_valid_o.write(true);
302  if(w_user_o.get_interface()) // optional user interface
303  w_user_o->write(ext->get_user(common::id_type::DATA));
304 
305  if(write_trans->phase == tlm::BEGIN_REQ)
306  w_last_o.write(true);
307 
308  write_trans->phase =
309  (write_trans->phase == axi::BEGIN_PARTIAL_REQ) ? axi::END_PARTIAL_REQ : tlm::END_REQ;
310  SCCTRACE(SCMOD) << write_trans->phase << " AW bit assignment trans: " << p << " addr: " << p->get_address()
311  << " axi_id: " << ext->get_id();
312  break;
313  }
314  }
315 
316  // The Slave puts the requested data on the Read Data channel and asserts RVALID,
317  // indicating the data in the channel is valid.
318  if(r_valid_i.read()) {
319  unsigned id = r_id_i.read();
320  auto it = active_r_transactions.find(id);
321  if(it == active_r_transactions.end())
322  SCCFATAL(SCMOD) << "Invalid read transaction ID " << id;
323 
324  auto read_trans = it->second.front();
325  if(read_trans->payload == nullptr)
326  SCCERR(SCMOD) << "Invalid transaction";
327 
328  if(read_trans->phase == tlm::END_REQ || read_trans->phase == axi::BEGIN_PARTIAL_RESP) {
329  auto data = r_data_i.read();
330  for(size_t j = 0, k = 0; k < BUSWIDTH / 8; j += 8, ++k) {
331  *(uint8_t*)(read_trans->payload->get_data_ptr() + k) = data.range(j + 7, j).to_uint();
332  }
333 
334  // The Master asserts RREADY, indicating the master is ready to receive data from the slave.
335  r_ready_o.write(true);
336  read_trans->phase = axi::BEGIN_PARTIAL_RESP;
337  if(r_last_i.read()) {
338  read_trans->phase = tlm::BEGIN_RESP;
339  }
340  if(r_user_i.get_interface()) { // optional user interface
341  payload_type* p = read_trans->payload;
342  auto ext = p->get_extension<axi::axi4_extension>();
343  sc_assert(ext && "axi4_extension missing");
344  ext->set_user(common::id_type::DATA, r_user_i->read());
345  }
346 
347  auto ret = input_socket->nb_transport_bw(*read_trans->payload, read_trans->phase, delay);
348  SCCTRACE(SCMOD) << read_trans->phase << " R bit assignment trans " << std::hex << read_trans->payload << std::dec
349  << " (axi_id:" << id << ")";
350 
351  // EDN_RESP indicates the last phase of the AXI Read transaction
352  if(ret == tlm::TLM_UPDATED && read_trans->phase == tlm::END_RESP) {
353  remove_trans(id, read_trans->payload->is_read());
354  if(read_trans->payload->has_mm())
355  read_trans->payload->release();
356  }
357  }
358  }
359 
360  if(b_valid_i.read()) {
361  unsigned id = b_id_i.read();
362  auto it = active_w_transactions.find(id);
363  if(it == active_w_transactions.end())
364  SCCFATAL(SCMOD) << "Invalid transaction ID " << id;
365 
366  auto write_trans = it->second.front();
367  if(write_trans->payload == nullptr)
368  SCCERR(SCMOD) << "Invalid transaction";
369 
370  if(write_trans->phase == tlm::END_REQ) {
371  sc_assert(b_resp_i.read() == axi::to_int(axi::resp_e::OKAY));
372  if(b_user_i.get_interface()) { // optional user interface
373  payload_type* p = write_trans->payload;
374  auto ext = p->get_extension<axi::axi4_extension>();
375  sc_assert(ext && "axi4_extension missing");
376  ext->set_user(common::id_type::RESP, b_user_i->read());
377  }
378  b_ready_o.write(true);
379  write_trans->phase = tlm::BEGIN_RESP;
380  auto ret = input_socket->nb_transport_bw(*write_trans->payload, write_trans->phase, delay);
381  SCCTRACE(SCMOD) << write_trans->phase << " B bit assignment trans " << std::hex << write_trans->payload << std::dec
382  << " (axi_id:" << id << ")";
383  if(ret == tlm::TLM_UPDATED && write_trans->phase == tlm::END_RESP) {
384  remove_trans(id, write_trans->payload->is_read());
385  }
386  }
387  }
388  }
389 }

391 template <unsigned int BUSWIDTH, unsigned int ADDRWIDTH, unsigned int IDWIDTH, unsigned int USERWIDTH>
392 void axi_tlm2pin_adaptor<BUSWIDTH, ADDRWIDTH, IDWIDTH, USERWIDTH>::remove_trans(const unsigned id, const bool is_read) {
393  if(is_read) {
394  auto it = active_r_transactions.find(id);
395  if(it == active_r_transactions.end())
396  SCCFATAL(SCMOD) << "Invalid read transaction ID " << id;
397 
398  it->second.pop();
399  if(it->second.empty())
400  active_r_transactions.erase(id);
401  } else {
402  auto it = active_w_transactions.find(id);
403  if(it == active_w_transactions.end())
404  SCCFATAL(SCMOD) << "Invalid write transaction ID " << id;
405 
406  it->second.pop();
407  if(it->second.empty())
408  active_w_transactions.erase(id);
409  }
410 }
