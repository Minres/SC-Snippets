121 template <typename T> inline void range_lut<T>::addEntry(T i, uint64_t base_addr, uint64_t size) {
122  auto iter = m_lut.find(base_addr);
123  if(iter != m_lut.end() && iter->second.index != null_entry)
124  throw std::runtime_error("range already mapped");
125 
126  auto eaddr = base_addr + size - 1;
127  if(eaddr < base_addr)
128  throw std::runtime_error("address wrap-around occurred");
129 
130  m_lut[base_addr] = lut_entry{i, size > 1 ? BEGIN_RANGE : SINGLE_BYTE_RANGE};
131  if(size > 1)
132  m_lut[eaddr] = lut_entry{i, END_RANGE};
133  ++m_size;
134 }

136 template <typename T> inline bool range_lut<T>::removeEntry(T i) {
137  auto start = m_lut.begin();
138  while(start->second.index != i && start != m_lut.end())
139  start++;
140  if(start != m_lut.end()) {
141  if(start->second.type == SINGLE_BYTE_RANGE) {
142  m_lut.erase(start);
143  } else {
144  auto end = start;
145  end++;
146  end++;
147  m_lut.erase(start, end);
148  }
149  --m_size;
150  return true;
151  }
152  return false;
153 }

155 template <typename T> inline void range_lut<T>::validate() const {
156  auto mapped = false;
157  for(auto iter = m_lut.begin(); iter != m_lut.end(); iter++) {
158  switch(iter->second.type) {
159  case SINGLE_BYTE_RANGE:
160  if(iter->second.index != null_entry && mapped)
161  throw std::runtime_error("range overlap: begin range while in mapped range");
162 
163  break;
164  case BEGIN_RANGE:
165  if(iter->second.index != null_entry) {
166  if(mapped) {
167  throw std::runtime_error("range overlap: begin range while in mapped range");
168  }
169  mapped = true;
170  }
171  break;
172  case END_RANGE:
173  if(!mapped) {
174  throw std::runtime_error("range overlap: end range while in unmapped region");
175  }
176  mapped = false;
177  break;
178  }
179  }
