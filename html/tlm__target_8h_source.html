128 template <unsigned int BUSWIDTH>
129 void scc::tlm_target<BUSWIDTH>::b_tranport_cb(tlm::tlm_generic_payload& gp, sc_core::sc_time& delay) {
130  resource_access_if* ra = nullptr;
131  uint64_t base = 0;
132  std::tie(ra, base) = socket_map.getEntry(gp.get_address());
133  if(ra) {
134  gp.set_response_status(tlm::TLM_BURST_ERROR_RESPONSE);
135  if(gp.get_data_length() <= ra->size()) {
136  gp.set_response_status(tlm::TLM_BYTE_ENABLE_ERROR_RESPONSE);
137  if(gp.get_byte_enable_ptr() == nullptr) {
138  gp.set_response_status(tlm::TLM_GENERIC_ERROR_RESPONSE);
139  if(gp.get_data_length() == gp.get_streaming_width()) {
140  if(gp.get_command() == tlm::TLM_READ_COMMAND) {
141  if(ra->read(gp.get_data_ptr(), gp.get_data_length(), (gp.get_address() - base), delay))
142  gp.set_response_status(tlm::TLM_OK_RESPONSE);
143  } else if(gp.get_command() == tlm::TLM_WRITE_COMMAND) {
144  if(ra->write(gp.get_data_ptr(), gp.get_data_length(), (gp.get_address() - base), delay))
145  gp.set_response_status(tlm::TLM_OK_RESPONSE);
146  } else {
147  gp.set_response_status(tlm::TLM_COMMAND_ERROR_RESPONSE);
148  }
149  }
150  }
151  }
152  } else {
153  gp.set_response_status(tlm::TLM_ADDRESS_ERROR_RESPONSE);
154  }
155  delay += clk;
156 }

158 template <unsigned int BUSWIDTH> unsigned int scc::tlm_target<BUSWIDTH>::tranport_dbg_cb(tlm::tlm_generic_payload& gp) {
159  resource_access_if* ra = nullptr;
160  uint64_t base = 0;
161  std::tie(ra, base) = socket_map.getEntry(gp.get_address());
162  if(ra) {
163  if(gp.get_data_length() == ra->size() && gp.get_byte_enable_ptr() == nullptr &&
164  gp.get_data_length() == gp.get_streaming_width()) {
165  if(gp.get_command() == tlm::TLM_READ_COMMAND) {
166  if(ra->read_dbg(gp.get_data_ptr(), gp.get_data_length(), (gp.get_address() - base) / ra->size()))
167  return gp.get_data_length();
168  } else if(gp.get_command() == tlm::TLM_READ_COMMAND) {
169  if(ra->write_dbg(gp.get_data_ptr(), gp.get_data_length(), (gp.get_address() - base) / ra->size()))
170  return gp.get_data_length();
171  }
172  }
173  }
174  return 0;
175 }
