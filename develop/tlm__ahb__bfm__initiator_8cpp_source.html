42 template <unsigned WIDTH> inline void initiator<WIDTH>::bus_task() {
43  data_t data{0};
44  while(true) {
45  wait(inqueue.get_event());
46  while(auto* trans = inqueue.get_next_transaction()) {
47  addr_phase.lock();
48  auto* ext = trans->template get_extension<ahb_extension>();
49  HADDR_o.write(trans->get_address());
50  HWRITE_o.write(trans->is_write());
51  HMASTLOCK_o.write(ext->is_locked());
52  HPROT_o.write(ext->get_protection());
53  HBURST_o.write(static_cast<unsigned>(ext->get_burst()));
54  auto log_bytes = scc::ilog2(trans->get_data_length());
55  auto log_width = scc::ilog2(WIDTH / 8);
56  size_t size = 0;
57  for(; size < log_width; ++size)
58  if(trans->get_address() & (1 << size))
59  break; // i contains the first bit not being 0
60  HSIZE_o.write(size);
61  HTRANS_o.write(static_cast<unsigned>(trans_e::NONSEQ));
62  wait(HCLK_i.posedge_event());
63  data_phase.lock();
64  tlm::tlm_phase phase{tlm::END_REQ};
65  sc_time delay;
66  tsckt->nb_transport_bw(*trans, phase, delay);
67  addr_phase.unlock();
68  auto beats = 1 << (log_bytes - size);
69  auto offset = 0U;
70  for(size_t i = 0; i < beats; ++i) {
71  if(beats == 1)
72  HTRANS_o.write(static_cast<unsigned>(trans_e::IDLE));
73  else if(i > 0)
74  HTRANS_o.write(static_cast<unsigned>(trans_e::SEQ));
75  if(trans->is_write()) {
76  for(size_t j = 0, k = 0; k < WIDTH / 8; j += 8, ++k, ++offset)
77  data.range(j + 7, j) = *(uint8_t*)(trans->get_data_ptr() + offset);
78  HWDATA_o.write(data);
79  }
80  wait(HCLK_i.posedge_event());
81  while(!HREADY_i.read())
82  wait(HCLK_i.posedge());
83  if(trans->is_read()) {
84  data = HRDATA_i.read();
85  for(size_t j = 0, k = 0; k < WIDTH / 8; j += 8, ++k, ++offset)
86  *(uint8_t*)(trans->get_data_ptr() + offset) = data.range(j + 7, j).to_uint();
87  }
88  }
89  phase = tlm::BEGIN_RESP;
90  delay = SC_ZERO_TIME;
91  tsckt->nb_transport_bw(*trans, phase, delay);
92  if(trans->has_mm())
93  trans->release();
94  HTRANS_o.write(static_cast<unsigned>(trans_e::IDLE));
95  data_phase.unlock();
96  }
97  }
98 }
