126 std::vector<std::string> scanModule(sc_core::sc_object const* obj, Module *currentModule, unsigned const level) {
127  SCCDEBUG() << indent*level<< obj->name() << "(" << obj->kind() << ")";
128  std::string kind{obj->kind()};
129  if (kind == "sc_module") {
130  auto* name = obj->name();
131  currentModule->submodules.push_back(Module(name, obj->basename(), type(*obj), false));
132  std::unordered_set<std::string> keep_outs;
133  for (auto* child : obj->get_child_objects()) {
134  const std::string child_name{child->basename()};
135  if(child_name.substr(0, 3)=="$$$")
136  continue;
137  if(!keep_outs.empty()) {
138  auto it = std::find_if(std::begin(keep_outs), std::end(keep_outs), [&child_name](std::string const& e){
139  return child_name.size() > e.size() && child_name.substr(0, e.size()) == e;
140  });
141  if(it!=std::end(keep_outs))
142  continue;
143  }
144  auto ks = scanModule(child, &currentModule->submodules.back(), level+1);
145  if(ks.size())
146  for(auto& s:ks) keep_outs.insert(s);
147  }
148  } else if(kind == "sc_clock"){
149  auto const* iface = dynamic_cast<sc_core::sc_interface const*>(obj);
150  currentModule->submodules.push_back(Module(obj->name(), obj->basename(), type(*obj), false));
151  currentModule->submodules.back().ports.push_back(Port(std::string(obj->name())+"."+obj->basename(), obj->basename(), iface, false, obj->kind()));
152 #ifndef NO_TLM_EXTRACT
153  } else if(auto const* tptr = dynamic_cast<tlm::tlm_base_socket_if const*>(obj)) {
154  auto cat = tptr->get_socket_category();
155  bool input = (cat & tlm::TLM_TARGET_SOCKET) == tlm::TLM_TARGET_SOCKET;
156  currentModule->ports.push_back(Port(obj->name(), obj->basename(), input?GET_EXPORT_IF(tptr):GET_PORT_IF(tptr), input, obj->kind()));
157  return {std::string(obj->basename())+"_port", std::string(obj->basename())+"_export"};
158 #endif
159  } else if (auto const* optr = dynamic_cast<sc_core::sc_port_base const*>(obj)) {
160  if(std::string(optr->basename()).substr(0, 3)!="$$$") {
161  sc_core::sc_interface const* pointer = optr->get_interface();
162  bool input = kind == "sc_in" || kind == "sc_fifo_in";
163  currentModule->ports.push_back(Port(obj->name(), obj->basename(), pointer, input, obj->kind()));
164  }
165  } else if (auto const* optr = dynamic_cast<sc_core::sc_export_base const*>(obj)) {
166  if(std::string(optr->basename()).substr(0, 3)!="$$$") {
167  sc_core::sc_interface const* pointer = optr->get_interface();
168  currentModule->ports.push_back(Port(obj->name(), obj->basename(), pointer, true, obj->kind()));
169  }
170  } else if (know_entities.find(std::string(obj->kind())) == know_entities.end()) {
171  SCCWARN() << "object not known (" << std::string(obj->kind()) << ")";
172  }
173  return {};
174 }

176 void generateElk(std::ostream& e, Module const& module, unsigned level=0) {
177  SCCDEBUG() << module.name;
178  unsigned num_in{0}, num_out{0};
179  for (auto port : module.ports) if(port.input) num_in++; else num_out++;
180  if(!module.ports.size() && !module.submodules.size()) return;
181  e << indent*level << "node " << module.name << " {" << "\n";
182  level++;
183  e << indent*level << "layout [ size: 50, "<<std::max(80U, std::max(num_in, num_out)*20)<<" ]\n";
184  e << indent*level << "portConstraints: FIXED_SIDE\n";
185  e << indent*level << "label \"" << module.name << "\"\n";
186 
187  for (auto port : module.ports) {
188  SCCDEBUG() << " " << port.name << "\n";
189  auto side = port.input?"WEST":"EAST";
190  e << indent*level << "port " << port.name << " { ^port.side: "<<side<<" label '" << port.name<< "' }\n";
191  }
192 
193  for (auto m : module.submodules)
194  generateElk(e, m, level);
195  // Draw edges module <-> submodule:
196  for (auto srcport : module.ports) {
197  if(srcport.port_if)
198  for (auto tgtmod : module.submodules) {
199  for (auto tgtport : tgtmod.ports) {
200  if (tgtport.port_if == srcport.port_if)
201  e << indent*level << "edge " << srcport.id << " -> " << tgtport.id << "\n";
202  }
203  }
204  }
205  // Draw edges submodule -> submodule:
206  for (auto srcmod : module.submodules) {
207  for (auto srcport : srcmod.ports) {
208  if(!srcport.input && srcport.port_if)
209  for (auto tgtmod : module.submodules) {
210  for (auto tgtport : tgtmod.ports) {
211  if(srcmod.id == tgtmod.id && tgtport.id == srcport.id)
212  continue;
213  if (tgtport.port_if == srcport.port_if && tgtport.input)
214  e << indent*level << "edge " << srcport.id << " -> " << tgtport.id << "\n";
215  }
216  }
217  }
218  }
219  level--;
220  e << indent*level << "}\n" << "\n";
221 }
222 void generateModJson(writer_type& writer, Module const& module, unsigned level=0) {
223  writer.StartObject();
224  writer.Key("id");
225  writer.String(module.id.c_str());
226  writer.Key("name");
227  writer.String(module.name.c_str());
228  writer.Key("type");
229  writer.String(module.type.c_str());
230  writer.Key("topmodule");
231  writer.Bool(module.topModule);
232  if(module.ports.size()) {
233  writer.Key("ports");
234  writer.StartArray();
235  for(auto& p: module.ports) {
236  writer.StartObject();
237  writer.Key("id");
238  writer.String(p.id.c_str());
239  writer.Key("name");
240  writer.String(p.name.c_str());
241  writer.Key("type");
242  writer.String(p.type.c_str());
243  writer.Key("input");
244  writer.Bool(p.input);
245  writer.Key("interface");
246  writer.Uint64(reinterpret_cast<uintptr_t>(p.port_if));
247  writer.EndObject();
248  }
249  writer.EndArray();
250  }
251  if(module.submodules.size()) {
252  writer.Key("children");
253  writer.StartArray();
254  for(auto& c: module.submodules) {
255  generateModJson(writer, c, level*1);
256  }
257  writer.EndArray();
258  }
259  writer.EndObject();
260 }

300 void dump_structure(std::ostream& e, hierarchy_dumper::file_type format) {
301  std::vector<Module> topModules;
302  std::vector<sc_core::sc_object*> tops = sc_core::sc_get_top_level_objects();
303  for (auto t : tops) {
304  if (std::string(t->kind()) == "sc_module" && std::string(t->basename()).substr(0, 3) != "$$$") {
305  SCCDEBUG() << t->name() << "(" << t->kind() << ")";
306  topModules.push_back(Module(t->name(), t->basename(), type(*t), true));
307  for (auto* child : t->get_child_objects())
308  scanModule(child, &topModules.back(), 1);
309  }
310  }
311  if(format == hierarchy_dumper::ELKT) {
312  e<<"algorithm: org.eclipse.elk.layered\n";
313  e<<"edgeRouting: ORTHOGONAL\n";
314  for (auto module : topModules)
315  generateElk(e, module);
316  SCCINFO() << "SystemC Structure Dumped to ELK file";
317  } else if(format==hierarchy_dumper::JSON) {
318  OStreamWrapper stream(e);
319  writer_type writer(stream);
320  if(topModules.size()==1) {
321  generateModJson(writer, topModules.front());
322  } else {
323  writer.StartObject();
324  writer.Key("children");
325  writer.StartArray();
326  for(auto& c:topModules) {
327  generateModJson(writer, c);
328  }
329  writer.EndArray();
330  writer.EndObject();
331  }
332  } else if(format==hierarchy_dumper::D3JSON) {
333  OStreamWrapper stream(e);
334  writer_type writer(stream);
335  if(topModules.size()==1) {
336  generateD3Json(writer, topModules.front());
337  } else {
338  writer.StartObject();
339  writer.Key("children");
340  writer.StartArray();
341  for(auto& c:topModules) {
342  generateD3Json(writer, c);
343  }
344  writer.EndArray();
345  writer.EndObject();
346  }
347  }
