133  bool write(const uint8_t* data, size_t length, uint64_t offset = 0,
134  sc_core::sc_time d = sc_core::SC_ZERO_TIME) override {
135  assert("Access out of range" && offset + length <= sizeof(DATATYPE));
136  auto temp(storage);
137  auto beg = reinterpret_cast<uint8_t*>(&temp) + offset;
138  std::copy(data, data + length, beg);
139  if(wr_cb)
140  return wr_cb(*this, temp, d);
141  storage = (temp & wrmask) | (storage & ~wrmask);
142  return true;
143  }
155  bool read(uint8_t* data, size_t length, uint64_t offset = 0,
156  sc_core::sc_time d = sc_core::SC_ZERO_TIME) const override {
157  assert("Access out of range" && offset + length <= sizeof(DATATYPE));
158  auto temp(storage);
159  if(rd_cb) {
160  if(!rd_cb(*this, temp, d))
161  return false;
162  } else
163  temp &= rdmask;
164  auto beg = reinterpret_cast<uint8_t*>(&temp) + offset;
165  std::copy(beg, beg + length, data);
166  return true;
167  }
179  bool write_dbg(const uint8_t* data, size_t length, uint64_t offset = 0) override {
180  assert("Offset out of range" && offset == 0);
