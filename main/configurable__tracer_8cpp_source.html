64 void configurable_tracer::descend(const sc_core::sc_object* obj, bool trace) {
65  if(obj == this)
66  return;
67  const char* kind = obj->kind();
68  if((types_to_trace & trace_types::SIGNALS) == trace_types::SIGNALS && strcmp(kind, "tlm_signal") == 0) {
69  if(trace)
70  obj->trace(trf);
71  return;
72  } else if(strcmp(kind, "sc_vector") == 0) {
73  if(trace)
74  for(auto o : obj->get_child_objects())
75  descend(o, trace);
76  return;
77  } else if(strcmp(kind, "sc_module") == 0) {
78  auto trace_enable = get_trace_enabled(obj, default_trace_enable);
79  if(trace_enable)
80  obj->trace(trf);
81  for(auto o : obj->get_child_objects())
82  descend(o, trace_enable);
83  } else if(strcmp(kind, "sc_variable") == 0) {
84  if(trace && (types_to_trace & trace_types::VARIABLES) == trace_types::VARIABLES)
85  obj->trace(trf);
86  } else if(const auto* tr = dynamic_cast<const scc::traceable*>(obj)) {
87  if(tr->is_trace_enabled())
88  obj->trace(trf);
89  for(auto o : obj->get_child_objects())
90  descend(o, tr->is_trace_enabled());
91  } else if(trace)
92  try_trace(trf, obj, types_to_trace);
93 }

95 auto scc::configurable_tracer::get_trace_enabled(const sc_core::sc_object* obj, bool fall_back) -> bool {
96  auto* attr = obj->get_attribute(EN_TRACING_STR);
97  if(attr != nullptr && dynamic_cast<const sc_core::sc_attribute<bool>*>(attr) != nullptr) {
98  const auto* a = dynamic_cast<const sc_core::sc_attribute<bool>*>(attr);
99  return a->value;
100  } else {
101  std::string hier_name{obj->name()};
102  auto h = cci_broker.get_param_handle(hier_name.append("." EN_TRACING_STR));
103  if(h.is_valid())
104  return h.get_cci_value().get_bool();
105  }
106  return fall_back;
107 }

109 void configurable_tracer::augment_object_hierarchical(const sc_core::sc_object* obj) {
110  if(dynamic_cast<const sc_core::sc_module*>(obj) != nullptr || dynamic_cast<const scc::traceable*>(obj) != nullptr) {
111  auto* attr = obj->get_attribute(EN_TRACING_STR);
112  if(attr == nullptr ||
113  dynamic_cast<const sc_core::sc_attribute<bool>*>(attr) == nullptr) { // check if we have no sc_attribute
114  std::string hier_name{obj->name()};
115  hier_name += "." EN_TRACING_STR;
116  auto h = cci_broker.get_param_handle(hier_name);
117  if(!h.is_valid()) // we have no cci_param so create one
118  params.push_back(new cci::cci_param<bool>(hier_name, default_trace_enable, "", cci::CCI_ABSOLUTE_NAME,
119  cci_originator));
120  }
121  for(auto* o : obj->get_child_objects())
122  augment_object_hierarchical(o);
123  }
124 }
