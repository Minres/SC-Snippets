195 template <typename Range, typename Value = typename Range::value_type>
196 std::string join(Range const& elements, char const* const delimiter) {
197  std::ostringstream os;
198  auto b = std::begin(elements);
199  auto e = std::end(elements);
200  if(b != e) {
201  std::copy(b, std::prev(e), std::ostream_iterator<Value>(os, delimiter));
202  b = std::prev(e);
203  }
204  if(b != e)
205  os << *b;
206  return os.str();
207 }
