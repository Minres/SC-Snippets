100 template <unsigned int BUSWIDTH, unsigned int ADDRWIDTH>
101 inline tlm::tlm_sync_enum axi4lite_tlm2pin_adaptor<BUSWIDTH, ADDRWIDTH>::nb_transport_fw(payload_type& trans,
102  phase_type& phase,
103  sc_core::sc_time& t) {
104  if(trans.has_mm())
105  trans.acquire();
106 
107  SCCTRACE(SCMOD) << phase << " of " << (trans.is_read() ? "RD" : "WR") << " forward trans " << std::hex << &trans;
108  tlm::tlm_sync_enum status{tlm::TLM_ACCEPTED};
109 
110  if(phase == axi::BEGIN_PARTIAL_REQ || phase == tlm::BEGIN_REQ) {
111  auto ext = trans.get_extension<axi::axi4_extension>();
112  sc_assert(ext && "axi4_extension missing");
113 
114  if(trans.is_read()) {
115  register_trans(trans, phase);
116  } else { // Write transaction
117  if(active_w_transactions.empty())
118  register_trans(trans, phase);
119  else {
120  auto act_trans = active_w_transactions.front();
121  if(act_trans->payload == &trans) // if transaction is ongoing update the phase
122  act_trans->phase = phase;
123  else // otherwise push transaction in the queue
124  register_trans(trans, phase);
125  }
126  }
127  } else if(phase == tlm::END_RESP) {
128  trans.set_response_status(tlm::TLM_OK_RESPONSE);
129  if(trans.is_read())
130  active_r_transactions.pop();
131  else
132  active_w_transactions.pop();
133 
134  if(trans.has_mm())
135  trans.release();
136  status = tlm::TLM_ACCEPTED;
137  }
138  return status;
139 }

141 template <unsigned int BUSWIDTH, unsigned int ADDRWIDTH>
142 inline void axi4lite_tlm2pin_adaptor<BUSWIDTH, ADDRWIDTH>::bus_thread() {
143  auto delay = sc_core::SC_ZERO_TIME;
144 
145  if(!resetn_i.read()) { // active-low reset
146  ar_valid_o.write(false);
147  aw_valid_o.write(false);
148  w_valid_o.write(false);
149  b_ready_o.write(false);
150  r_ready_o.write(false);
151  } else {
152  if(aw_ready_i.read())
153  aw_valid_o.write(false);
154  if(w_ready_i.read()) {
155  w_valid_o.write(false);
156  }
157  if(ar_ready_i.read()) {
158  ar_valid_o.write(false);
159  }
160  r_ready_o.write(false);
161  b_ready_o.write(false);
162 
163  // AR channel
164  if(!active_r_transactions.empty()) {
165  auto read_trans = active_r_transactions.front();
166  if(read_trans->phase == tlm::BEGIN_REQ) {
167  payload_type* p = read_trans->payload;
168  auto ext = p->get_extension<axi::axi4_extension>();
169  sc_assert(ext && "axi4_extension missing");
170 
171  ar_addr_o.write(p->get_address());
172  ar_prot_o.write(ext->get_prot());
173  ar_valid_o.write(true);
174 
175  read_trans->phase = tlm::END_REQ;
176  SCCTRACE(SCMOD) << read_trans->phase << " RD BW trans: " << p << " addr: " << p->get_address();
177  input_socket->nb_transport_bw(*read_trans->payload, read_trans->phase, delay);
178  }
179  }
180 
181  // AW+W channel
182  if(!active_w_transactions.empty()) {
183  auto write_trans = active_w_transactions.front();
184  if(write_trans->phase == axi::BEGIN_PARTIAL_REQ || write_trans->phase == tlm::BEGIN_REQ) {
185  payload_type* p = write_trans->payload;
186  auto ext = p->get_extension<axi::axi4_extension>();
187  sc_assert(ext && "axi4_extension missing");
188 
189  if(write_trans->beat_cnt == 0) { // TODO: check if required for axi lite
190  aw_addr_o.write(p->get_address());
191  aw_prot_o.write(ext->get_prot());
192  aw_valid_o.write(true);
193  }
194  write_trans->beat_cnt++;
195 
196  // write data
197  sc_dt::sc_biguint<BUSWIDTH> data{0};
198  for(size_t j = 0, k = 0, offset = 0; k < 32 / 8; j += 8, ++k, ++offset)
199  data.range(j + 7, j) = *(uint8_t*)(p->get_data_ptr() + offset);
200 
201  w_strb_o.write(p->get_byte_enable_length());
202  w_data_o.write(data);
203  w_valid_o.write(true);
204 
205  write_trans->phase =
206  (write_trans->phase == axi::BEGIN_PARTIAL_REQ) ? axi::END_PARTIAL_REQ : tlm::END_REQ;
207  SCCTRACE(SCMOD) << write_trans->phase << " RD BW trans: " << p << " addr: " << p->get_address();
208  input_socket->nb_transport_bw(*write_trans->payload, write_trans->phase, delay);
209  }
210  }
211 
212  // The Slave puts the requested data on the Read Data channel and asserts RVALID,
213  // indicating the data in the channel is valid.
214  if(r_valid_i.read()) {
215  if(!active_r_transactions.empty()) {
216  auto read_trans = active_r_transactions.front();
217  if(read_trans->payload == nullptr)
218  SCCERR(SCMOD) << "Invalid transaction";
219 
220  if(read_trans->phase == tlm::END_REQ) {
221  auto data = r_data_i.read();
222  for(size_t j = 0, k = 0; k < BUSWIDTH / 8; j += 8, ++k) {
223  *(uint8_t*)(read_trans->payload->get_data_ptr() + k) = data.range(j + 7, j).to_uint();
224  }
225 
226  // The Master asserts RREADY, indicating the master is ready to receive data from the slave.
227  r_ready_o.write(true);
228  read_trans->phase = tlm::BEGIN_RESP;
229 
230  auto ret = input_socket->nb_transport_bw(*read_trans->payload, read_trans->phase, delay);
231  SCCTRACE(SCMOD) << read_trans->phase << " bw trans " << std::hex << read_trans->payload;
232 
233  // EDN_RESP indicates the last phase of the AXI Read transaction
234  if(ret == tlm::TLM_UPDATED && read_trans->phase == tlm::END_RESP) {
235  active_r_transactions.pop();
236  if(read_trans->payload->has_mm())
237  read_trans->payload->release();
238  }
239  }
240  }
241  }
242 
243  if(b_valid_i.read()) {
244  if(!active_w_transactions.empty()) {
245  auto write_trans = active_w_transactions.front();
246  if(write_trans->payload == nullptr)
247  SCCERR(SCMOD) << "Invalid transaction";
248 
249  if(write_trans->phase == tlm::END_REQ) {
250  sc_assert(b_resp_i.read() == axi::to_int(axi::resp_e::OKAY));
251  b_ready_o.write(true);
252  write_trans->phase = tlm::BEGIN_RESP;
253  auto ret = input_socket->nb_transport_bw(*write_trans->payload, write_trans->phase, delay);
254  SCCTRACE(SCMOD) << write_trans->phase << " bw trans " << std::hex << write_trans->payload;
255  if(ret == tlm::TLM_UPDATED && write_trans->phase == tlm::END_RESP) {
256  active_w_transactions.pop();
257  }
258  }
259  }
260  }
261  }
