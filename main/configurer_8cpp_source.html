109  void dump_config(sc_core::sc_object* obj, writer_type& writer) {
110  auto start = std::string(obj->basename()).substr(0, 3);
111  if(start == "$$$") return;
112  auto obj_started = false;
113  for(sc_core::sc_attr_base* attr_base : obj->attr_cltn()) {
114  obj_started = start_object(writer, obj->basename(), obj_started);
115  check_n_assign<int>(writer, attr_base) || check_n_assign<unsigned>(writer, attr_base) ||
116  check_n_assign<long>(writer, attr_base) || check_n_assign<unsigned long>(writer, attr_base) ||
117  check_n_assign<long long>(writer, attr_base) || check_n_assign<unsigned long long>(writer, attr_base) ||
118  check_n_assign<bool>(writer, attr_base) || check_n_assign<float>(writer, attr_base) ||
119  check_n_assign<double>(writer, attr_base) || check_n_assign<std::string>(writer, attr_base) ||
120  check_n_assign<char*>(writer, attr_base) || check_n_assign<sc_core::sc_time>(writer, attr_base);
121  }
122 #ifdef HAS_CCI
123  const std::string hier_name{obj->name()};
124  auto log_lvl_set = false;
125  auto it = lut.find(obj->name());
126  if(it != lut.end())
127  for(auto& h : it->second) {
128  obj_started = start_object(writer, obj->basename(), obj_started);
129  auto value = h.get_cci_value();
130  std::string paramname{h.name()};
131  auto basename = paramname.substr(paramname.rfind('.') + 1);
132  if(basename == SCC_LOG_LEVEL_PARAM_NAME)
133  log_lvl_set = true;
134  if(value.is_bool())
135  writeValue(writer, basename, (bool)value.get_bool());
136  else if(value.is_int())
137  writeValue(writer, basename, (int)value.get_int());
138  else if(value.is_int64())
139  writeValue(writer, basename, static_cast<int64_t>(value.get_int64()));
140  else if(value.is_uint())
141  writeValue(writer, basename, value.get_uint());
142  else if(value.is_uint64())
143  writeValue(writer, basename, static_cast<uint64_t>(value.get_uint64()));
144  else if(value.is_double())
145  writeValue(writer, basename, value.get_double());
146  else if(value.is_string())
147  writeValue(writer, basename, value.get_string().c_str());
148  }
149  auto mod = dynamic_cast<sc_core::sc_module*>(obj);
150  if(!log_lvl_set && mod) {
151  obj_started = start_object(writer, obj->basename(), obj_started);
152  auto val = broker.get_preset_cci_value(fmt::format("{}.{}", obj->name(), SCC_LOG_LEVEL_PARAM_NAME));
153  auto global_verb = static_cast<int>(get_logging_level());
154  writeValue(writer, "log_level", val.is_int() ? val.get_int() : global_verb);
155  }
156 #endif
157  for(auto* o : get_sc_objects(obj)) {
158  obj_started = start_object(writer, obj->basename(), obj_started);
159  dump_config(o, writer);
160  }
161  if(obj_started)
162  writer.EndObject();
163  }
164 };

175 void try_set_value(sc_core::sc_attr_base* attr_base, Value const& hier_val) {
176  CHECK_N_ASSIGN(int, attr_base, hier_val.Get<int>());
177  CHECK_N_ASSIGN(unsigned, attr_base, hier_val.Get<unsigned>());
178  CHECK_N_ASSIGN(int64_t, attr_base, hier_val.Get<int64_t>());
179  CHECK_N_ASSIGN(uint64_t, attr_base, hier_val.Get<uint64_t>());
180  CHECK_N_ASSIGN(bool, attr_base, hier_val.Get<bool>());
181  CHECK_N_ASSIGN(float, attr_base, hier_val.Get<float>());
182  CHECK_N_ASSIGN(double, attr_base, hier_val.Get<double>());
183  CHECK_N_ASSIGN(std::string, attr_base, std::string(hier_val.GetString()));
184  CHECK_N_ASSIGN(char*, attr_base, strdup(hier_val.GetString()));
185  {
186  auto* a = dynamic_cast<sc_core::sc_attribute<sc_core::sc_time>*>(attr_base);
187  if(a != nullptr) {
188  a->value = sc_core::sc_time(hier_val.Get<double>(), sc_core::SC_SEC);
189  return;
190  }
191  }

254 void configure_sc_attribute_hierarchical(Value const& node, std::string const& prefix) {
255  for(auto itr = node.MemberBegin(); itr != node.MemberEnd(); ++itr) {
256  if(!itr->name.IsString())
257  return;
258  auto key_name = itr->name.GetString();
259  if(strncmp(key_name, "log_level", 9) == 0)
260  continue; // virtual attribute
261  auto hier_name = prefix.size() ? prefix + "." + key_name : key_name;
262  Value const& val = itr->value;
263  if(val.IsNull() || val.IsArray())
264  continue;
265  else if(val.IsObject()) {
266  if(sc_core::sc_find_object(hier_name.c_str())) {
267  configure_sc_attribute_hierarchical(val, hier_name);
268  }
269  } else {
270  auto pos = hier_name.rfind('.');
271  if(pos != std::string::npos) {
272  auto objname = hier_name.substr(0, pos);
273  auto attrname = hier_name.substr(pos + 1);
274  if(auto* obj = sc_core::sc_find_object(objname.c_str()))
275  if(auto attr = obj->get_attribute(attrname.c_str()))
276  try_set_value(attr, val);
277  }
278  }
279  }

282 void check_config_hierarchical(configurer::broker_t const& broker, Value const& node, std::string const& prefix) {
283  for(auto itr = node.MemberBegin(); itr != node.MemberEnd(); ++itr) {
284  if(!itr->name.IsString())
285  return;
286  auto key_name = itr->name.GetString();
287  if(strncmp(key_name, SCC_LOG_LEVEL_PARAM_NAME, 9) == 0)
288  continue; // virtual attribute
289  auto hier_name = prefix.size() ? prefix + "." + key_name : key_name;
290  Value const& val = itr->value;
291  if(val.IsNull() || val.IsArray())
292  continue;
293  else if(val.IsObject()) {
294  if(!sc_core::sc_find_object(hier_name.c_str())) {
295  if(prefix.size())
296  throw std::domain_error(hier_name);
297  } else
298  check_config_hierarchical(broker, val, hier_name);
299  } else {
300  auto pos = hier_name.rfind('.');
301  if(pos != std::string::npos) {
302  auto objname = hier_name.substr(0, pos);
303  auto attrname = hier_name.substr(pos + 1);
304  auto* obj = sc_core::sc_find_object(objname.c_str());
305  if(!obj || !obj->get_attribute(attrname.c_str())) {
306 #ifdef HAS_CCI
307  auto param_handle = broker.get_param_handle(hier_name);
308  if(!param_handle.is_valid() && attrname != SCC_LOG_LEVEL_PARAM_NAME)
309 #endif
310  throw std::invalid_argument(hier_name);
311  }
312  }
313  }
314  }

334  try {
335  IStreamWrapper stream(is);
336  root->document.ParseStream(stream);
337  if(root->document.HasParseError()) {
338  SCCERR() << "Could not parse input file " << filename << ", location "
339  << (unsigned)root->document.GetErrorOffset()
340  << ", reason: " << GetParseError_En(root->document.GetParseError());
341  } else {
342  config_reader reader(cci_broker);
343  reader.configure_cci_hierarchical(root->document, "");
344  config_valid = true;
345  }
346  } catch(std::runtime_error& e) {
347  SCCERR() << "Could not parse input file " << filename << ", reason: " << e.what();
348  }
349  } else {
350  SCCERR() << "Could not open input file " << filename;
351  }
352  }
353 }

385 auto get_value_from_hierarchy(const std::string& hier_name, Value const& value) -> Value const& {
386  size_t npos = hier_name.find_first_of('.');
387  auto member = value.FindMember(hier_name.substr(0, npos).c_str());
388  auto& val = member->value;

390  return val;
391  return get_value_from_hierarchy(hier_name.substr(npos + 1, hier_name.size()), val);
