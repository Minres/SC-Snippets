160 std::ostream& operator<<(std::ostream& os, const tlm::tlm_generic_payload& t) {
161  char const* ch =
162  t.get_command() == tlm::TLM_READ_COMMAND ? "AR" : (t.get_command() == tlm::TLM_WRITE_COMMAND ? "AW" : "");
163  os << "CMD:" << cmd_str[t.get_command()] << ", " << ch << "ADDR:0x" << std::hex << t.get_address() << ", TXLEN:0x"
164  << t.get_data_length();
165  if(auto e = t.get_extension<axi::ace_extension>()) {
166  os << ", " << ch << "ID:" << e->get_id() << ", " << ch << "LEN:0x" << std::hex
167  << static_cast<unsigned>(e->get_length()) << ", " << ch << "SIZE:0x" << static_cast<unsigned>(e->get_size())
168  << std::dec << ", " << ch << "BURST:" << to_char(e->get_burst()) << ", " << ch
169  << "PROT:" << static_cast<unsigned>(e->get_prot()) << ", " << ch
170  << "CACHE:" << static_cast<unsigned>(e->get_cache()) << ", " << ch
171  << "QOS:" << static_cast<unsigned>(e->get_qos()) << ", " << ch
172  << "REGION:" << static_cast<unsigned>(e->get_region()) << ", " << ch << "SNOOP:0x" << std::hex
173  << (static_cast<unsigned>(e->get_snoop()) & 0xf) << std::dec << ", " << ch
174  << "DOMAIN:" << to_char(e->get_domain()) << ", " << ch << "BAR:" << to_char(e->get_barrier()) << ", " << ch
175  << "UNIQUE:" << e->get_unique();
176  } else if(auto e = t.get_extension<axi::axi4_extension>()) {
177  os << ", " << ch << "ID:" << e->get_id() << ", " << ch << "LEN:0x" << std::hex
178  << static_cast<unsigned>(e->get_length()) << ", " << ch << "SIZE:0x" << static_cast<unsigned>(e->get_size())
179  << std::dec << ", " << ch << "BURST:" << to_char(e->get_burst()) << ", " << ch
180  << "PROT:" << static_cast<unsigned>(e->get_prot()) << ", " << ch
181  << "CACHE:" << static_cast<unsigned>(e->get_cache()) << ", " << ch
182  << "QOS:" << static_cast<unsigned>(e->get_qos()) << ", " << ch
183  << "REGION:" << static_cast<unsigned>(e->get_region());
184  } else if(auto e = t.get_extension<axi::axi3_extension>()) {
185  os << ", " << ch << "ID:" << e->get_id() << ", " << ch << "LEN:0x" << std::hex
186  << static_cast<unsigned>(e->get_length()) << ", " << ch << "SIZE:0x" << static_cast<unsigned>(e->get_size())

191  << "REGION:" << static_cast<unsigned>(e->get_region());

235 template <> char const* is_valid_msg<axi::ace_extension>(axi::ace_extension* ext) {
236  auto offset = to_int(ext->get_snoop());
237  if(offset < 32) { // a read access
238  if(!rd_valid[offset & 0xf][to_int(ext->get_domain())])
239  return "illegal read snoop value";
240  } else {
241  if(!wr_valid[offset & 0xf][to_int(ext->get_domain())])
242  return "illegal write snoop value";
243  }
244  // check table ED3-7 and D3-8 of IHI0022H
245  switch(ext->get_snoop()) {
246  case snoop_e::READ_NO_SNOOP:
247  case snoop_e::WRITE_NO_SNOOP: // non coherent access to coherent domain
248  if(ext->get_domain() != domain_e::NON_SHAREABLE && ext->get_domain() != domain_e::SYSTEM) {
249  return "illegal domain for no non-coherent access";
250  }
251  break;
252  case snoop_e::READ_ONCE:
253  case snoop_e::READ_SHARED:
254  case snoop_e::READ_CLEAN:
255  case snoop_e::READ_NOT_SHARED_DIRTY:
256  case snoop_e::READ_UNIQUE:
257  case snoop_e::MAKE_UNIQUE:
258  case snoop_e::DVM_COMPLETE:
259  case snoop_e::DVM_MESSAGE:
260  case snoop_e::WRITE_UNIQUE:
261  case snoop_e::WRITE_LINE_UNIQUE:
262  case snoop_e::EVICT:
263  if(ext->get_domain() != domain_e::INNER_SHAREABLE && ext->get_domain() != domain_e::OUTER_SHAREABLE) {
264  return "illegal domain for coherent access";
265  }
266  break;
267  case snoop_e::CLEAN_SHARED:
268  case snoop_e::CLEAN_INVALID:
269  case snoop_e::MAKE_INVALID:
270  case snoop_e::WRITE_CLEAN:
271  case snoop_e::WRITE_BACK:
272  case snoop_e::WRITE_EVICT:
273  if(ext->get_domain() == domain_e::SYSTEM) {
274  return "illegal domain for coherent access";
275  }
276  break;
277  default:
278  break;
279  }
280  if((ext->get_barrier() == bar_e::MEMORY_BARRIER || ext->get_barrier() == bar_e::SYNCHRONISATION_BARRIER) &&
281  (offset & 0xf) != 0)
282  return "illegal barrier/snoop value combination";
283  switch(ext->get_cache()) {
284  case 4:
285  case 5:
286  case 8:
287  case 9:
288  case 12:
289  case 13:
290  return "illegal cache value";
291  default:
292  break;
293  }
294  if((ext->get_cache() & 2) == 0 && ext->get_domain() != domain_e::NON_SHAREABLE &&
295  ext->get_domain() != domain_e::SYSTEM)
296  return "illegal domain for no non-cachable access";
297  return nullptr;
298 }
