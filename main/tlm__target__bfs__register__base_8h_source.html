57 template <typename datatype_t>
58 class abstract_bitfield {
59  public:
60  constexpr abstract_bitfield(std::string name, size_t bitOffset,
61  size_t bitSize, std::string urid)
62  : bitOffset{bitOffset},
63  bitSize{bitSize},
64  name{std::move(name)},
65  urid{std::move(urid)} {}
66 
67  virtual void write(datatype_t& valueToWrite) = 0;
68  virtual datatype_t read() = 0;
69 
70  constexpr bool affected(size_t byteOffset, size_t byteLength) const noexcept {
71  return (byteOffset * 8 < bitOffset + bitSize) &&
72  (bitOffset < (byteOffset + byteLength) * 8);
73  }
74 
75  constexpr datatype_t mask() const noexcept {
76  return ((1ULL << bitSize) - 1) << bitOffset;
77  }
78 
79  const size_t bitOffset;
80  const size_t bitSize;
81  const std::string name;
82  const std::string urid;
83 };

106  datatype_t resetValue = 0,
107  datatype_t writeMask = -1,
108  datatype_t readMask = -1)
109  : sc_core::sc_object{name},
110  offset{offset},
111  resetValue{resetValue},
112  writeMask{writeMask},

114 
118  constexpr size_t size() const noexcept override { return sizeof(datatype_t); }
119 
120  void reset() override { storage = resetValue; }
121 
122  bool write(const uint8_t* data, std::size_t length, uint64_t offset = 0,
123  sc_core::sc_time d = sc_core::SC_ZERO_TIME) override {
124  assert("Access out of range" && offset + length <= this->size());
125  auto valueToWrite{storage};
126  std::copy(data, data + length,
127  reinterpret_cast<uint8_t*>(&valueToWrite) + offset);
128  for (auto&& bitfield : bitfields) {
129  if (bitfield.get().affected(offset, length)) {
130  auto mask = bitfield.get().mask();
131  auto bits = (valueToWrite & mask) >> bitfield.get().bitOffset;
132  bitfield.get().write(bits);
133  valueToWrite = (valueToWrite & ~mask) |
134  ((bits << bitfield.get().bitOffset) & mask);
135  }
136  }
137  if (writeCallback) writeCallback(*this, valueToWrite);
138  storage = (valueToWrite & writeMask) | (storage & ~writeMask);
139  return true;
140  }
141 
142  bool read(uint8_t* data, std::size_t length, uint64_t offset = 0,
143  sc_core::sc_time d = sc_core::SC_ZERO_TIME) const override {
144  assert("Access out of range" && offset + length <= this->size());
145  auto result{storage};
146  result &= readMask;
147  for (auto&& bitfield : bitfields) {
148  if (bitfield.get().affected(offset, length)) {
149  auto bitfieldValue = bitfield.get().read();
150  auto mask = bitfield.get().mask();
151  result = (result & ~mask) |
152  ((bitfieldValue << bitfield.get().bitOffset) & mask);
153  }
154  }
155  if (readCallback) readCallback(*this, result);
156  auto begin = reinterpret_cast<const uint8_t*>(&result) + offset;
157  std::copy(begin, begin + length, data);
158  return true;
159  }
160 
161  bool write_dbg(const uint8_t* data, std::size_t length,
162  uint64_t offset = 0) override {
163  assert("Offset out of range" && offset == 0);
164  if (length != this->size()) return false;
165  std::copy(data, data + length, reinterpret_cast<uint8_t*>(&storage));
166  return true;
167  }
168 
169  bool read_dbg(uint8_t* data, std::size_t length,

171  assert("Offset out of range" && offset == 0);
172  if (length != this->size()) return false;
173  auto storagePtr = reinterpret_cast<const uint8_t*>(&storage);
174  std::copy(storagePtr, storagePtr + length, data);
175  return true;
176  }

249  public:

253 
263  // constexpr //TODO: requires c++14
264  bitfield(bitfield_register<datatype_t>& reg, std::string name,
265  size_t bitOffset, size_t bitSize, std::string urid,
266  Access access = RW)
267  : reg{reg},
268  abstract_bitfield<datatype_t>{std::move(name), bitOffset, bitSize,
269  std::move(urid)},
270  access{access} {
271  reg.registerBitfield(*this);
272  }
273  bitfield(const bitfield&) = delete;
274  bitfield& operator=(const bitfield&) = delete;
275 
276  void write(datatype_t& valueToWrite) override {
277  if (writeCallback) writeCallback(*this, valueToWrite);
278  if (access == ReadOnly) valueToWrite = get();
279  }
280  datatype_t read() override {
281  if (readCallback)
282  return readCallback(*this);
283  else
284  return get();
285  }
286 
290  constexpr datatype_t get() const { return (reg.get() & mask()) >> bitOffset; }
294  void put(datatype_t value) {
295  reg.put((reg.get() & ~mask()) | ((value << bitOffset) & mask()));
296  }
297 
301  constexpr operator datatype_t() const { return get(); }
305  bitfield<datatype_t>& operator=(datatype_t other) {
306  put(other);
307  return *this;
308  }

335  readCallback = std::move(callback);
336  }

339  Access access;
340 
341  protected:

374  bitfield_register<uint32_t>& getRegister(const std::string& name) {
375  auto found =
376  std::find_if(asDerived().registers.begin(), asDerived().registers.end(),
377  [name](const bitfield_register<uint32_t>& reg) {
378  return name.compare(reg.basename()) == 0;
379  });
380  if (found == asDerived().registers.end()){
381  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("Register "+ name + " not found").c_str());
382  }
383  return *found;
384  }

393  bitfield<uint32_t>& getBitfieldByName(const std::string& regname,
394  const std::string& name) {
395  auto found = std::find_if(
396  asDerived().bitfields.begin(), asDerived().bitfields.end(),
397  [regname, name](const bitfield<uint32_t>& bf) {
398  return regname.compare(bf.reg.basename()) == 0 && name == bf.name;
399  });
400  if (found == asDerived().bitfields.end()){
401  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("Bitfield " + name +
402  " in register " + regname + " not found").c_str());
403  }
404  return *found;
405  }

413  bitfield<uint32_t>& getBitfieldById(const std::string& urid) {
414  auto found = std::find_if(
415  asDerived().bitfields.begin(), asDerived().bitfields.end(),
416  [urid](const bitfield<uint32_t>& bf) { return urid == bf.urid; });
417  if (found == asDerived().bitfields.end()) {
418  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("Bitfield with urid " + urid + " not found").c_str());
419  }
420  return *found;
421  }
422 
433  bitfield<uint32_t>& getBitfield(const std::string& regname,
434  const std::string& name,
435  const std::string& urid) {
436  if (use_URID) {
437  return getBitfieldById(urid);
438  } else {
439  bitfield<uint32_t>& result = getBitfieldByName(regname, name);
440  if (result.urid != urid) {
441  SC_REPORT_WARNING(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("URID of register is " + result.urid + " but " + urid
442  + " was passed").c_str());
443  }
444  return result;
445  }
446  }
