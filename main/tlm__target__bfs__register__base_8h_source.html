57 template <typename datatype_t> class abstract_bitfield {
58 public:
59  constexpr abstract_bitfield(std::string name, size_t bitOffset, size_t bitSize, std::string urid)
60  : bitOffset{bitOffset}
61  , bitSize{bitSize}
62  , name{std::move(name)}
63  , urid{std::move(urid)} {}
64 
65  virtual void write(datatype_t& valueToWrite) = 0;
66  virtual datatype_t read() = 0;
67 
68  constexpr bool affected(size_t byteOffset, size_t byteLength) const noexcept {
69  return (byteOffset * 8 < bitOffset + bitSize) && (bitOffset < (byteOffset + byteLength) * 8);
70  }
71 
72  constexpr datatype_t mask() const noexcept { return ((1ULL << bitSize) - 1) << bitOffset; }
73 
74  const size_t bitOffset;
75  const size_t bitSize;
76  const std::string name;
77  const std::string urid;
78 };

85 template <typename datatype_t> class bitfield_register : public sc_core::sc_object, public scc::resource_access_if {
86 public:
98  constexpr bitfield_register(sc_core::sc_module_name name, size_t offset, datatype_t resetValue = 0,
99  datatype_t writeMask = -1, datatype_t readMask = -1)
100  : sc_core::sc_object{name}
101  , offset{offset}
102  , resetValue{resetValue}
103  , writeMask{writeMask}
104  , readMask{readMask} {}
105 
109  constexpr size_t size() const noexcept override { return sizeof(datatype_t); }
110 
111  void reset() override { storage = resetValue; }
112 
113  bool write(const uint8_t* data, std::size_t length, uint64_t offset = 0,
114  sc_core::sc_time d = sc_core::SC_ZERO_TIME) override {
115  assert("Access out of range" && offset + length <= this->size());
116  auto valueToWrite{storage};
117  std::copy(data, data + length, reinterpret_cast<uint8_t*>(&valueToWrite) + offset);
118  for(auto&& bitfield : bitfields) {
119  if(bitfield.get().affected(offset, length)) {
120  auto mask = bitfield.get().mask();
121  auto bits = (valueToWrite & mask) >> bitfield.get().bitOffset;
122  bitfield.get().write(bits);
123  valueToWrite = (valueToWrite & ~mask) | ((bits << bitfield.get().bitOffset) & mask);
124  }
125  }
126  if(writeCallback)
127  writeCallback(*this, valueToWrite);
128  storage = (valueToWrite & writeMask) | (storage & ~writeMask);
129  return true;
130  }
131 
132  bool read(uint8_t* data, std::size_t length, uint64_t offset = 0,
133  sc_core::sc_time d = sc_core::SC_ZERO_TIME) const override {
134  assert("Access out of range" && offset + length <= this->size());
135  auto result{storage};
136  result &= readMask;
137  for(auto&& bitfield : bitfields) {
138  if(bitfield.get().affected(offset, length)) {
139  auto bitfieldValue = bitfield.get().read();
140  auto mask = bitfield.get().mask();
141  result = (result & ~mask) | ((bitfieldValue << bitfield.get().bitOffset) & mask);
142  }
143  }
144  if(readCallback)
145  readCallback(*this, result);
146  auto begin = reinterpret_cast<const uint8_t*>(&result) + offset;
147  std::copy(begin, begin + length, data);
148  return true;
149  }
150 
151  bool write_dbg(const uint8_t* data, std::size_t length, uint64_t offset = 0) override {
152  assert("Offset out of range" && offset == 0);
153  if(length != this->size())
154  return false;
155  std::copy(data, data + length, reinterpret_cast<uint8_t*>(&storage));
156  return true;
157  }
158 
159  bool read_dbg(uint8_t* data, std::size_t length, uint64_t offset = 0) const override {
160  assert("Offset out of range" && offset == 0);
161  if(length != this->size())
162  return false;
163  auto storagePtr = reinterpret_cast<const uint8_t*>(&storage);
164  std::copy(storagePtr, storagePtr + length, data);
165  return true;
166  }
167 
171  constexpr datatype_t get() const { return storage; }
175  void put(datatype_t value) { storage = value; }
176 
177  void registerBitfield(abstract_bitfield<datatype_t>& bitfield) { bitfields.push_back(bitfield); }
178 
182  constexpr operator datatype_t() const { return storage; }
186  bitfield_register<datatype_t>& operator=(datatype_t other) {
187  storage = other;
188  return *this;
189  }

203  writeCallback = std::move(callback);

213  readCallback = std::move(callback);
214  }

216  const size_t offset;
217 
218 protected:
219  const datatype_t resetValue;
220  const datatype_t writeMask;
221  const datatype_t readMask;
222  datatype_t storage;

227 };

229 template <typename datatype_t> class bitfield : public abstract_bitfield<datatype_t> {
230 public:
231  enum Access { RW, ReadOnly };
232  using abstract_bitfield<datatype_t>::mask;
233  using abstract_bitfield<datatype_t>::bitOffset;
234 
244  // constexpr //TODO: requires c++14
245  bitfield(bitfield_register<datatype_t>& reg, std::string name, size_t bitOffset, size_t bitSize, std::string urid,
246  Access access = RW)
247  : reg{reg}
248  , abstract_bitfield<datatype_t>{std::move(name), bitOffset, bitSize, std::move(urid)}
249  , access{access} {
250  reg.registerBitfield(*this);
251  }
252  bitfield(const bitfield&) = delete;
253  bitfield& operator=(const bitfield&) = delete;
254 
255  void write(datatype_t& valueToWrite) override {
256  if(writeCallback)
257  writeCallback(*this, valueToWrite);
258  if(access == ReadOnly)
259  valueToWrite = get();
260  }
261  datatype_t read() override {
262  if(readCallback)
263  return readCallback(*this);
264  else
265  return get();
266  }
267 
271  constexpr datatype_t get() const { return (reg.get() & mask()) >> bitOffset; }
275  void put(datatype_t value) { reg.put((reg.get() & ~mask()) | ((value << bitOffset) & mask())); }
276 
280  constexpr operator datatype_t() const { return get(); }
284  bitfield<datatype_t>& operator=(datatype_t other) {
285  put(other);
286  return *this;
287  }
288 
300  void setWriteCallback(std::function<void(bitfield<datatype_t>&, datatype_t& valueToWrite)> callback) {
301  writeCallback = std::move(callback);
302  }
310  void setReadCallback(std::function<datatype_t(const bitfield<datatype_t>&)> callback) {
311  readCallback = std::move(callback);
312  }
313 
314  bitfield_register<datatype_t>& reg;
315  Access access;
316 
317 protected:
318  std::function<void(bitfield<datatype_t>&, datatype_t&)> writeCallback;
319  std::function<datatype_t(const bitfield<datatype_t>&)> readCallback;
320 };

348  bitfield_register<uint32_t>& getRegister(const std::string& name) {
349  auto found =
350  std::find_if(asDerived().registers.begin(), asDerived().registers.end(),
351  [name](const bitfield_register<uint32_t>& reg) { return name.compare(reg.basename()) == 0; });
352  if(found == asDerived().registers.end()) {
353  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("Register " + name + " not found").c_str());
354  }
355  return *found;
356  }

365  bitfield<uint32_t>& getBitfieldByName(const std::string& regname, const std::string& name) {
366  auto found = std::find_if(asDerived().bitfields.begin(), asDerived().bitfields.end(),
367  [regname, name](const bitfield<uint32_t>& bf) {
368  return regname.compare(bf.reg.basename()) == 0 && name == bf.name;
369  });
370  if(found == asDerived().bitfields.end()) {
371  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE,
372  ("Bitfield " + name + " in register " + regname + " not found").c_str());
373  }
374  return *found;
375  }
376 
383  bitfield<uint32_t>& getBitfieldById(const std::string& urid) {
384  auto found = std::find_if(asDerived().bitfields.begin(), asDerived().bitfields.end(),
385  [urid](const bitfield<uint32_t>& bf) { return urid == bf.urid; });
386  if(found == asDerived().bitfields.end()) {
387  SC_REPORT_FATAL(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE, ("Bitfield with urid " + urid + " not found").c_str());
388  }
389  return *found;
390  }

402  bitfield<uint32_t>& getBitfield(const std::string& regname, const std::string& name, const std::string& urid) {
403  if(use_URID) {
404  return getBitfieldById(urid);
405  } else {
406  bitfield<uint32_t>& result = getBitfieldByName(regname, name);
407  if(result.urid != urid) {
408  SC_REPORT_WARNING(ID_SCC_TLM_TARGET_BFS_REGISTER_BASE,
409  ("URID of register is " + result.urid + " but " + urid + " was passed").c_str());
410  }
411  return result;
412  }
413  }
