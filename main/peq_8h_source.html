157  void insert_entry(const TYPE &entry, sc_core::sc_time abs_time) {
158  auto it = m_scheduled_events.find(abs_time);
159  if (it == m_scheduled_events.end()) {
160  if (free_pool.size()) {
161  auto r = m_scheduled_events.insert(std::make_pair(abs_time, free_pool.front()));
162  free_pool.pop_front();

170  }
